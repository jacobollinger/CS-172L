    public static boolean colleague(Scientist[] a){
        for(int i = 0; i < a.length; i++){
            for(int j = i + 1; j < a.length - 1; i++){
                if(a[i].yearOfBirth == a[j].yearOfBirth && a[i].fieldOfStudy == a[j].fieldOfStudy){
                    return true;
                }
            }
        }
        return false;
    }
